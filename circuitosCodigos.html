Pisca LED

Circuito:

Código:

por escrito:

int pinLed = 12;

void setup(){

pinMode(pinLed, OUTPUT);

}

void loop(){

digitalWrite(pinLed, HIGH);

delay(1000);

digitalWrite(pinLed, LOW);

delay(1000);

}

Sensor

Circuito:

Código:

por escrito:

int pinLed = 13;

int sensor = 7;

int valor;

void setup(){

pinMode(pinLed, OUTPUT);

pinMode(sensor, INPUT);

}

void loop(){

valor = digitalRead(sensor);

if(valor == HIGH){

digitalWrite(pinLed, HIGH);

}else{

digitalWrite(pinLed, LOW);

}

}
